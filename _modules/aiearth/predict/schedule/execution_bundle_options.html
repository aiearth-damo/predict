[文档]    def to_dict(self):
        d = {}
        if self.num_cpus:
            d["num_cpus"] = self.num_cpus
        if self.num_gpus:
            if d.get("num_cpus", 0) != 0:
                raise RuntimeError(
                    "It is not allowed to specify both num_cpus and num_gpus"
                )
            d["num_gpus"] = self.num_gpus
        if self.memory:
            d["memory"] = self.memory
        if self.batch_size:
            d["batch_size"] = self.batch_size
        return d


    def __str__(self):
